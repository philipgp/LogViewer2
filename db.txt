
CREATE TABLE LOG_METADATA
(
ID INT AUTO_INCREMENT  PRIMARY KEY ,  
LOG_DATE  DATE ,
FULL_TEXT VARCHAR(255),
LOG_MSG VARCHAR(255),
THREAD_NAME VARCHAR(255),
LEVEL VARCHAR(255),  
TRANSACTION_ID NUMBER,
);



CREATE TABLE METADATA
(
ID INT AUTO_INCREMENT  PRIMARY KEY ,  
KEY VARCHAR(255),
VALUE VARCHAR(255),
LOG_METADATA_ID INT 
);
